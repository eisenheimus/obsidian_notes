```bash
# -d (detach) - запуск контейнера в фоновом режиме, не блокирует консоль
docker run -d \
# --name - задаёт имя контейнера для удобного управления
  --name prometheus \ 
# --restart unless-stopped - политика перезапуска:
#   перезапускать контейнер автоматически, если он остановился,
#   ЕСЛИ только он не был остановлен вручную командой docker stop
  --restart unless-stopped \
# --network - подключает контейнер к указанной сети
#   позволяет общаться с другими контейнерами по имени
  --network monitoring \
# -p (publish) - проброс портов: порт_хоста:порт_контейнера
#   делает порт 9090 контейнера доступным на хосте
  -p 9090:9090 \
# -v (volume) - монтирование тома/директории: путь_на_хосте:путь_в_контейнере
#   слева - директория на хосте (сохраняется при удалении контейнера)
#   справа - куда монтируется внутри контейнера
  -v /etc/prometheus:/etc/prometheus \
  -v /var/lib/prometheus:/prometheus \
# Образ, на основе которого создаётся контейнер
#   prom/prometheus - официальный образ Prometheus
#   :latest - тег версии (последняя стабильная)
  prom/prometheus:latest
```

```bash
--restart no           # никогда не перезапускать (по умолчанию)
--restart always       # всегда перезапускать, даже если остановлен вручную
--restart on-failure   # перезапускать только при ошибке (код выхода не 0)
--restart unless-stopped # перезапускать всегда, КРОМЕ случаев ручной остановки
```

```bash
# После образа идут аргументы, которые передаются в Prometheus
  prom/prometheus:latest \
  --config.file=/etc/prometheus/prometheus.yml \
  # путь к файлу конфигурации (внутри контейнера)
  
  --storage.tsdb.path=/prometheus \
  # путь для хранения данных TSDB (Time Series Database)
```
