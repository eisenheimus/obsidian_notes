
# Смена имени контейнера
 <br>
 
```bash
# 100 - id контейнера
pct set 100 -hostname новое-имя-100

# Альтернативный спосбо смены имени
# Войдите в контейнер
pct enter 100

# Измените имя хоста
echo "новое-имя" > /etc/hostname
hostname -F /etc/hostname

# Список контейнеров с новыми именами
pct list
```
