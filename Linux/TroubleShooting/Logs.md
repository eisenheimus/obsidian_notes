```bash
# просмотр логов в реальном времени
tail -f /var/logs/nginx.log

# Просмотр последних 10 строк логов
tail -n 10 /var/logs/nginx.log

# просмотр ВСЕХ логов в реальном времени
journalctl -f 

# Просмотр логов сервиса
journalctl -u nginx.servie


# 0 - неработоспособность системы
# 1 - Предупреждения немедленно выполнить
# 2 - Критическое состояние 
# 3 - Ошибки
# 4 - Предупреждения
# 5 - Только уведомления
# 6 - Информационные сообщения
# 7 - Отладочная информация
journalctl -p [1-7]
```
